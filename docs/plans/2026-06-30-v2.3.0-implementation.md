### 3c. Add `animateNaturalReflow()` FLIP function

```typescript
/**
 * FLIP 动画：让块级插入后的"下方块被推下去"视觉上平滑过渡。
 * First: 快照所有块位置 → Invert: 反推到旧位置 → Play: 取消 transform 平滑过渡。
 *
 * 位移幅度 = 块实际被推的距离（编辑器原生 reflow delta），不做 opacity。
 */
function animateNaturalReflow(editor: HTMLElement): void {
  const now = performance.now();
  // 跳过条件：距离上次块级变化 < 100ms（连续 Enter / 快速 paste）
  if (now - lastBlockInsertTime < 100) return;
  lastBlockInsertTime = now;

  // First：快照所有块位置
  const first = new Map<HTMLElement, number>();
  const blockEls = editor.querySelectorAll<HTMLElement>('[data-node-id]');
  blockEls.forEach(el => {
    first.set(el, el.getBoundingClientRect().top);
  });

  // 让浏览器重新布局（此时 DOM 已变），下一帧读新位置
  requestAnimationFrame(() => {
    for (const [el, y0] of first) {
      const y1 = el.getBoundingClientRect().top;
      const delta = y0 - y1; // 正 = 被推下去
      if (Math.abs(delta) < 2) continue; // 微动跳过

      // Invert：推到旧位置
      el.style.transform = `translateY(${delta}px)`;
      el.style.transition = 'none';

      // 强制 reflow（让 Invert 立即生效）
      void el.offsetHeight;

      // Play：平滑过渡回原位
      requestAnimationFrame(() => {
        el.style.transition = `transform 250ms ${SCROLL_CURVE}`;
        el.style.transform = '';
        // 动画结束后清理 inline style
        setTimeout(() => {
          el.style.transition = '';
        }, 300);
      });
    }
  });
}
```

```typescript
/**
 * v2.3.0：把块按 .?!。？！ 切成句，返回句子包裹的 span 元素数组。
 * 用 TreeWalker 遍历文本节点找切分点，用 Range + extractContents + insertNode 包裹。
 *
 * 关键技术决策（librarian 调研结论，见 docs/research/2026-06-30-siyuan-editor-dom.md）：
 * - 不用 range.surroundContents()，因为它会在跨 inline 元素（如 strong/em/a）时抛错
 * - 改用 range.extractContents() + span.appendChild(fragment) + range.insertNode(span)
 * - span 用 position: absolute + visibility: hidden + pointer-events: none，避免影响布局
 * - 必须保存/恢复 window.getSelection()，否则光标位置会丢失
 * - 必须过滤 block type（跳过 code-block/AV/table/math-block/iframe），只处理 NodeParagraph/NodeHeading/NodeListItem
 */
const SENTENCE_SPAN_CLASS = 'zt-sentence-span';

// block type 白名单 — 只处理可读文本块，跳过 code/AV/table/math/iframe
const RIPPLE_TARGET_BLOCK_TYPES = new Set(['NodeParagraph', 'NodeHeading', 'NodeListItem']);

// 跳过：嵌入块、代码块、数据库视图、表格、数学块
const RIPPLE_SKIP_BLOCK_TYPES = new Set([
  'NodeCodeBlock',
  'NodeBlockQueryEmbed',
  'NodeAttributeView',
  'NodeTable',
  'NodeMathBlock',
]);

function isRippleTargetBlock(block: HTMLElement): boolean {
  const type = block.dataset.type;
  if (!type) return false;
  if (RIPPLE_SKIP_BLOCK_TYPES.has(type)) return false;
  return RIPPLE_TARGET_BLOCK_TYPES.has(type);
}

function getSentences(block: HTMLElement): HTMLElement[] {
  if (!isRippleTargetBlock(block)) return [];

  const text = block.textContent ?? '';
  if (!text) return [];

  // 先清理旧句级 span（如果上次 applyRipple 后未清理）
  removeSentenceSpans(block);

  const sentences: HTMLElement[] = [];
  const regex = /[.?!。？！]+/g;
  const walker = document.createTreeWalker(block, NodeFilter.SHOW_TEXT);
  const textNodes: Text[] = [];
  let node: Text | null;
  while ((node = walker.nextNode() as Text)) textNodes.push(node);

  if (textNodes.length === 0) return [];

  // 累积文本位置映射
  let globalOffset = 0;
  const offsets = textNodes.map(tn => {
    const start = globalOffset;
    globalOffset += tn.length;
    return { node: tn, start, end: globalOffset };
  });

  // 保存当前光标位置
  const sel = window.getSelection();
  const savedRange = sel && sel.rangeCount > 0 ? sel.getRangeAt(0).cloneRange() : null;

  let lastIdx = 0;
  let match;
  while ((match = regex.exec(text)) !== null) {
    const matchEnd = match.index + match[0].length;
    if (matchEnd <= lastIdx) continue; // 重叠跳过

    // 包裹 [lastIdx, matchEnd) 作为一个句子
    const span = wrapTextRange(block, offsets, lastIdx, matchEnd);
    if (span) sentences.push(span);
    lastIdx = matchEnd;
  }

  // 尾段（如果还有剩余文本）
  if (lastIdx < text.length) {
    const span = wrapTextRange(block, offsets, lastIdx, text.length);
    if (span) sentences.push(span);
  }

  // 兜底：如果无法切句（如 inline 元素太密无法分句），返回整个块作为单句
  if (sentences.length === 0) {
    const span = document.createElement('span');
    span.className = SENTENCE_SPAN_CLASS;
    span.style.cssText = 'position: absolute; visibility: hidden; pointer-events: none;';
    span.textContent = text;
    sentences.push(span);
  }

  // 恢复光标位置
  if (savedRange && sel) {
    sel.removeAllRanges();
    sel.addRange(savedRange);
  }

  return sentences;
}
```

#### 4c. Add `wrapTextRange` helper

```typescript
/**
 * 在 block 中用 Range 包裹 [startOffset, endOffset) 范围的文本节点内容。
 * 返回创建的 span 元素，如果失败返回 null。
 *
 * 为什么不用 range.surroundContents()：
 * surroundContents 在跨 inline 元素（strong/em/a/block-ref）时会抛 BAD_BOUNDARYPOINTS_ERR。
 * SiYuan 块内通常有大量 inline 元素，这种错误是常态而非异常。
 * 改用 extractContents + insertNode 模式，参考 docs/research/2026-06-30-siyuan-editor-dom.md
 */
function wrapTextRange(
  block: HTMLElement,
  offsets: { node: Text; start: number; end: number }[],
  startOffset: number,
  endOffset: number
): HTMLElement | null {
  const range = document.createRange();

  // 找 start 所在节点
  let startFound = false;
  for (const off of offsets) {
    if (startOffset >= off.start && startOffset < off.end) {
      range.setStart(off.node, startOffset - off.start);
      startFound = true;
      break;
    }
  }
  if (!startFound) return null;

  // 找 end 所在节点
  let endFound = false;
  for (const off of offsets) {
    if (endOffset > off.start && endOffset <= off.end) {
      range.setEnd(off.node, endOffset - off.start);
      endFound = true;
      break;
    }
  }
  if (!endFound) return null;

  const span = document.createElement('span');
  span.className = SENTENCE_SPAN_CLASS;
  span.style.cssText = 'position: absolute; visibility: hidden; pointer-events: none;';

  try {
    // 提取范围内的所有节点（包括 inline 元素）
    const fragment = range.extractContents();
    span.appendChild(fragment);
    // 在原 range 位置插入 span（range 此时是空的，位于 start 处）
    range.insertNode(span);
    return span;
  } catch (err) {
    // 极端情况：跨 AV/嵌入块边界等
    return null;
  }
}
```

```typescript
/**
 * v2.3.0：找到当前光标所在句的 span 元素。
 * 基于 window.getSelection() 的 anchorNode，向上找 .zt-sentence-span。

function getCurrentSentence(block: HTMLElement): HTMLElement | null {
  if (block.textContent && block.textContent.length < 2) return block; // 短文本，整块就是"当前句"

  const sel = window.getSelection();
  if (!sel || !sel.rangeCount) return null;
  const range = sel.getRangeAt(0);
  if (!range.collapsed) return null;

  let node: Node | null = range.startContainer;
  while (node && node !== block) {
    if (node instanceof HTMLElement && node.classList.contains(SENTENCE_SPAN_CLASS)) {
      return node;
    }
    node = node.parentNode;
  }
  return null; // 未找到句级 span，退化为块级
}
```

```typescript
/**
 * 清理块内所有句级 span，恢复原始文本结构。
 * 在 clearAllOpacity 前调用。
 */
function removeSentenceSpans(block: HTMLElement): void {
  const spans = block.querySelectorAll(`.${SENTENCE_SPAN_CLASS}`);
  spans.forEach(span => {
    const parent = span.parentNode;
    if (!parent) return;
    while (span.firstChild) {
```

#### 5b. Add `visualWeightOf(node)` helper

Add after `calculateBlockDistance()`:

```typescript
/**
 * v2.3.0：块在视窗中的可见高度 / 编辑器可视区高度。
 * 用于列表项动态算法（Q5=C），weight=0 表示块完全在视窗外，1 表示占满可视区。
 */
function visualWeightOf(node: HTMLElement): number {
  const rect = node.getBoundingClientRect();
  const editor = node.closest<HTMLElement>('.protyle-wysiwyg');
  if (!editor) return 1.0;
  const editorRect = editor.getBoundingClientRect();
  // 只算在编辑器内的可见部分
  const visibleTop = Math.max(rect.top, editorRect.top);
  const visibleBottom = Math.min(rect.bottom, editorRect.bottom);
  const visibleHeight = Math.max(0, visibleBottom - visibleTop);
  return Math.min(1, visibleHeight / (editorRect.height || 1));
}
```

#### 5c. Add `depthOf(node)` helper

```typescript
/**
 * v2.3.0：列表深度 — 数 data-subtype="o"|"t"|"u" 的祖先元素个数。
 * depth=0 表示根块，depth=3 表示三层嵌套列表。
 */
function depthOf(node: HTMLElement): number {
  let depth = 0;
  let el: HTMLElement | null = node;
  while (el && el !== document.body) {
    const subtype = el.dataset.subtype;
    if (subtype === 'o' || subtype === 't' || subtype === 'u') {
      depth++;
    }
    el = el.parentElement;
  }
  return depth;
}
```

#### 5d. Add `depthFactor(depth)` helper

```typescript
/**
 * v2.3.0：列表深度系数 — 每深一层 opacity × (1 - DEPTH_FACTOR)。
 * depth=0 → 1.0, depth=1 → 0.95, depth=2 → 0.90, ... 最低 floor=0.7。
 */
function depthFactorFromDepth(depth: number): number {
  return Math.max(0.7, 1.0 - depth * DEPTH_FACTOR);
}
```

**Note on naming**: Use `depthFactorFromDepth` to avoid name collision with `DEPTH_FACTOR` config constant. Alternatively, name the config constant `RIPPLE_DEPTH_FACTOR` in step 1 and keep the helper as `depthFactor(depth)`. For simplicity, use `calculateDepthFactor`:


#### 5e. Add `isEmbedBlock(node)` helper

```typescript
/**
 * v2.3.0：判断块是否为嵌入内容（iframe / video / PDF / 嵌入块）。
 * 嵌入内容视觉权重更低，应用 EMBED_MULTIPLIER。
```

```typescript
function applyRipple(): void {
  if (pendingFrame !== null) return;
  pendingFrame = requestAnimationFrame(() => {
    pendingFrame = null;

    if (!inputMode.isFocusActive()) {
      clearAllOpacity();
      return;
    }

    if (shouldPauseFocusAndTypewriter()) {
      clearAllOpacity();
      return;
    }

    const currentBlock = getCurrentBlock();
    if (!currentBlock) return;

    const container = currentBlock.closest(".protyle-wysiwyg");
    if (!(container instanceof HTMLElement)) return;

    // v2.3.0：句级切分（getSentences 内部已清理旧 span）
    getSentences(currentBlock as HTMLElement); // 创建 .zt-sentence-span
    const currentSentence = getCurrentSentence(currentBlock as HTMLElement);

    const allBlocks = Array.from(
      container.querySelectorAll<HTMLElement>('[data-node-id], iframe, video')
    );

    allBlocks.forEach((block) => {
      const distance = calculateBlockDistance(currentBlock, block);
      // 句级梯度：distance=0 → 当前块，distance>0 → 按距离衰减
      const baseOpacity = SENTENCE_LEVELS[Math.min(distance, SENTENCE_LEVELS.length - 1)];

      let opacity = baseOpacity;

      // v2.3.0：视觉权重（列表项可见高度占比）
      const weight = visualWeightOf(block);
      opacity *= (WEIGHT_MIN + weight * (1 - WEIGHT_MIN)); // lerp(WEIGHT_MIN, 1.0, weight)

      // v2.3.0：嵌入块修正
      if (isEmbedBlock(block)) {
        opacity *= EMBED_MULTIPLIER;
      }

      // v2.3.0：列表深度系数
      const depth = depthOf(block);
      if (depth > 0) {
        opacity *= calculateDepthFactor(depth);
      }

      // v2.3.0：句级微调 — 距当前块 0 的块内，非当前句 × 0.88
      if (distance === 0 && currentSentence) {
        const allSentences = block.querySelectorAll<HTMLElement>(`.${SENTENCE_SPAN_CLASS}`);
        allSentences.forEach(s => {
          if (s !== currentSentence) {
            s.style.opacity = String(SENTENCE_LEVELS[1]); // 0.88
          } else {
            s.style.opacity = "1";
          }
        });
      }

      block.style.opacity = String(opacity);
    });

    // 当前句 / 当前块强制最亮
    if (currentSentence) currentSentence.style.opacity = "1";
    (currentBlock as HTMLElement).style.opacity = "1";
  });
}
```
